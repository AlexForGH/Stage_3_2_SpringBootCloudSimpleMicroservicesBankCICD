1. docker run --rm --name my-postgres -e POSTGRES_USER=user -e POSTGRES_PASSWORD=pass -e POSTGRES_DB=my_schema -p 5432:5432 -d postgres
        --rm	                            Контейнер удалится автоматически после остановки (включая Ctrl+C)
        --name my-postgres	                Имя контейнера (удобно для управления)
        -e POSTGRES_USER=user	            Создаёт пользователя user
        -e POSTGRES_PASSWORD=pass	        Устанавливает пароль
        -e POSTGRES_DB=my_schema	        Создаёт базу данных my_schema при первом запуске
        -p 5432:5432	                    Пробрасывает порт 5432 контейнера на хост
        -d 	                                Запускает образ в фоне
        -postgres	                        Имя официального запускаемого образа PostgreSQL

2. docker run -d -p 8080:8080 --name keycloak -e KC_BOOTSTRAP_ADMIN_USERNAME=admin -e KC_BOOTSTRAP_ADMIN_PASSWORD=admin quay.io/keycloak/keycloak:26.1.3 start-dev
